cmake_minimum_required(VERSION 3.15)
set(CMAKE_CXX_STANDARD 20)

project(KlemmgineGame)

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
add_subdirectory("../../" "Engine")

file(
	GLOB_RECURSE
	SRCS
	"Code/**.cpp"
)

add_executable(KlemmgineGame ${SRCS})

KlemmgineDependsOnBuildTool(
	KlemmgineGame
	"${CMAKE_BINARY_DIR}/GeneratedIncludes"
	"${CMAKE_CURRENT_LIST_DIR}/Code/Objects"
	"${CMAKE_CURRENT_LIST_DIR}/../../EngineSource/Objects"
)

target_link_libraries(KlemmgineGame Klemmgine)
