cmake_minimum_required(VERSION 3.15)
set(CMAKE_CXX_STANDARD 20)

project(TestProject)

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
include("../../Klemmgine.cmake")

file(
	GLOB_RECURSE
	SRCS
	"Code/**.cpp"
)

add_executable(TestProject ${SRCS})

KlemmgineDependsOnBuildTool(
	TestProject
	"${CMAKE_BINARY_DIR}/GeneratedIncludes"
	"${CMAKE_CURRENT_LIST_DIR}/Code/Objects"
	"${CMAKE_CURRENT_LIST_DIR}/../../EngineSource/Objects"
)

target_link_libraries(TestProject LibKlemmgine)
